Create Table GEN_STATUS (
ID BIGINT NOT NULL AUTO_INCREMENT,
UTC VARCHAR(255) ,
SYSTEMID TINYINT ,
GENREADYTOSTART VARCHAR(255) ,
GENENGINERUNNING VARCHAR(255) ,
GENCURRENTR VARCHAR(255) ,
GENCURRENTS VARCHAR(255) ,
GENCURRENTT VARCHAR(255) ,
GENVOLTAGE VARCHAR(255) ,
GENFREQ VARCHAR(255) ,
GENPOWER VARCHAR(255) , PRIMARY KEY(ID));
